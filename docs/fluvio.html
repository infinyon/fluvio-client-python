fluvio

The __fluvio__ python module provides an extension for working with the Fluvio streaming platform.

This module builds on top of the Fluvio Client Rust Crate and provides a pythonic access to the API.

Producing data to a topic in a Fluvio cluster is as simple as:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
producer = fluvio.topic_producer(topic)

for i in range(10):
    producer.send_string("Hello %s " % i)

Consuming is also simple:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
builder = ConsumerConfigExtBuilder(topic)
config = builder.build()
stream = fluvio.consumer_with_config(config)

num_items = 2
records = [bytearray(next(stream).value()).decode() for _ in range(num_items)]

For more examples see the integration tests in the fluvio-python repository.

182        kind: SmartModuleKind = None,
183        params: typing.Dict[str, str] = None,
184        aggregate: typing.List[bytes] = None,
185    ):
186        """
187        This is a method for adding a smartmodule to a consumer config either
188        using a `name` of a `SmartModule` or a `path` to a wasm binary.
189
190        Args:

201
202        Returns:
203
204            None
205        """

207        if kind is not None:
208            kind = kind.value
209
210        if path is None and name is None:
211            raise Exception("Require either a path or a name for a smartmodule.")
212
213        if path is not None and name is not None:
214            raise Exception("Only specify one of path or name not both.")
215
216        params = {} if params is None else params
217        param_keys = [x for x in params.keys()]
218        param_vals = [x for x in params.values()]
219
220        self._inner.smartmodule(
221            name,
222            path,
223            kind,
224            param_keys,
225            param_vals,
226            aggregate,
227            # These arguments are for Join stuff but that's not implemented on
228            # the python side yet
229            None,
230            None,
231            None,
232            None,

236class PartitionConsumer:
237    """
238    An interface for consuming events from a particular partition
239
240    There are two ways to consume events: by "fetching" events and by
241    "streaming" events. Fetching involves specifying a range of events that you
242    want to consume via their Offset. A fetch is a sort of one-time batch
243    operation: you’ll receive all of the events in your range all at once. When
244    you consume events via Streaming, you specify a starting Offset and receive
245    an object that will continuously yield new events as they arrive.
246    """

254        """
255        Continuously streams events from a particular offset in the consumer’s
256        partition. This returns a `Iterator[Record]` which is an
257        iterator.
258
259        Streaming is one of the two ways to consume events in Fluvio. It is a
260        continuous request for new records arriving in a partition, beginning
261        at a particular offset. You specify the starting point of the stream
262        using an Offset and periodically receive events, either individually or
263        in batches.
264        """
265        return self._generator(self._inner.stream(offset))

490class TopicProducer:
491    """An interface for producing events to a particular topic.
492
493    A `TopicProducer` allows you to send events to the specific topic it was
494    initialized for. Once you have a `TopicProducer`, you can send events to
495    the topic, choosing which partition each event should be delivered to.
496    """
497
498    _inner: _TopicProducer
499
500    def __init__(self, inner: _TopicProducer):
501        self._inner = inner
502
503    def send_string(self, buf: str) -> ProduceOutput:
504        """Sends a string to this producer’s topic"""
505        return self.send([], buf.encode("utf-8"))
506
507    async def async_send_string(self, buf: str) -> ProduceOutput:
508        """Sends a string to this producer’s topic"""
509        return await self.async_send([], buf.encode("utf-8"))
510
511    def send(self, key: typing.List[int], value: typing.List[int]) -> ProduceOutput:
512        """
513        Sends a key/value record to this producer's Topic.
514
515        The partition that the record will be sent to is derived from the Key.
516        """
517        return ProduceOutput(self._inner.send(key, value))
518
519    async def async_send(
520        self, key: typing.List[int], value: typing.List[int]
521    ) -> ProduceOutput:
522        """
523        Async sends a key/value record to this producer's Topic.
524
525        The partition that the record will be sent to is derived from the Key.
526        """
527        produce_output = await self._inner.async_send(key, value)
528        return ProduceOutput(produce_output)
529
530    def flush(self) -> None:
531        """
532        Send all the queued records in the producer batches.
533        """
534        return self._inner.flush()
535
536    async def async_flush(self) -> None:
537        """
538        Async send all the queued records in the producer batches.
539        """
540        await self._inner.async_flush()
541
542    def send_all(
543        self, records: typing.List[typing.Tuple[bytes, bytes]]
544    ) -> typing.List[ProduceOutput]:
545        """
546        Sends a list of key/value records as a batch to this producer's Topic.
547        :param records: The list of records to send
548        """
549        records_inner = [_ProducerBatchRecord(x, y) for (x, y) in records]
550        return [
551            ProduceOutput(output_inner)
552            for output_inner in self._inner.send_all(records_inner)
553        ]
554
555    async def async_send_all(
556        self, records: typing.List[typing.Tuple[bytes, bytes]]
557    ) -> typing.List[ProduceOutput]:
558        """
559        Async sends a list of key/value records as a batch to this producer's Topic.
560        :param records: The list of records to send
561        """
562        records_inner = [_ProducerBatchRecord(x, y) for (x, y) in records]
563        return [
564            ProduceOutput(output_inner)
565            for output_inner in await self._inner.async_send_all(records_inner)
566        ]
