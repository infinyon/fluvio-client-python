fluvio

The __fluvio__ python module provides an extension for working with the Fluvio streaming platform.

This module builds on top of the Fluvio Client Rust Crate and provides a pythonic access to the API.

Creating a topic with default settings is as simple as:

fluvio_admin = FluvioAdmin.connect()
fluvio_admin.create_topic("a_topic")

Or just create a topic with custom settings:

import fluvio

fluvio_admin = FluvioAdmin.connect()
topic_spec = (
    TopicSpec.create()
    .with_retention_time("1h")
    .with_segment_size("10M")
    .build()
)
fluvio_admin.create_topic("a_topic", topic_spec)

Producing data to a topic in a Fluvio cluster is as simple as:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
producer = fluvio.topic_producer(topic)

for i in range(10):
    producer.send_string("Hello %s " % i)

Consuming is also simple:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
builder = ConsumerConfigExtBuilder(topic)
config = builder.build()
stream = fluvio.consumer_with_config(config)

num_items = 2
records = [bytearray(next(stream).value()).decode() for _ in range(num_items)]

Producing with a custom configuration:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
builder = (TopicProducerConfigBuilder()
    .batch_size(32768)
    .linger(100)
    .compression(Compression.Gzip)
    .max_request_size(33554432)
    .timeout(600000)
    .isolation(Isolation.ReadCommitted)
    .delivery_semantic(DeliverySemantic.AtLeastOnce)
)
producer = fluvio.topic_producer_with_config(self.topic, config)

for i in range(10):
    producer.send_string("Hello %s " % i)

Also you can consume usign offset management:

import fluvio

fluvio = Fluvio.connect()

topic = "a_topic"
builder = ConsumerConfigExtBuilder(topic)
builder.offset_start(Offset.beginning())
builder.offset_strategy(OffsetManagementStrategy.MANUAL)
builder.offset_consumer("a-consumer")
config = builder.build()
stream = fluvio.consumer_with_config(config)

num_items = 2
records = [bytearray(next(stream).value()).decode() for _ in range(num_items)]

stream.offset_commit()
stream.offset_flush()

267        self,
268        name: str = None,
269        path: str = None,
270        kind: SmartModuleKind = None,
271        params: typing.Dict[str, str] = None,
272        aggregate: typing.List[bytes] = None,
273    ):
274        """
275        This is a method for adding a smartmodule to a consumer config either
276        using a `name` of a `SmartModule` or a `path` to a wasm binary.
277
278        Args:
279
280            name: str
281            path: str
282            kind: SmartModuleKind
283            params: Dict[str, str]
284            aggregate: List[bytes] # This is used for the initial value of an aggregate smartmodule
285
286        Raises:
287            "Require either a path or a name for a smartmodule."
288            "Only specify one of path or name not both."
289
290        Returns:
291
292            None
293        """
294
295        if kind is not None:
296            kind = kind.value
297
298        if path is None and name is None:
299            raise Exception("Require either a path or a name for a smartmodule.")
300
301        if path is not None and name is not None:
302            raise Exception("Only specify one of path or name not both.")
303
304        params = {} if params is None else params
305        param_keys = [x for x in params.keys()]
306        param_vals = [x for x in params.values()]
307
308        self._inner.smartmodule(
309            name,
310            path,
311            kind,
312            param_keys,
313            param_vals,
314            aggregate,
315            # These arguments are for Join stuff but that's not implemented on
316            # the python side yet
317            None,
318            None,
319            None,
320            None,
321        )

344class PartitionConsumer:
345    """
346    An interface for consuming events from a particular partition
347
348    There are two ways to consume events: by "fetching" events and by
349    "streaming" events. Fetching involves specifying a range of events that you
350    want to consume via their Offset. A fetch is a sort of one-time batch
351    operation: you’ll receive all of the events in your range all at once. When
352    you consume events via Streaming, you specify a starting Offset and receive
353    an object that will continuously yield new events as they arrive.
354    """
355
356    _inner: _PartitionConsumer
357
358    def __init__(self, inner: _PartitionConsumer):
359        self._inner = inner
360
361    def stream(self, offset: Offset) -> typing.Iterator[Record]:
362        """
363        Continuously streams events from a particular offset in the consumer’s
364        partition. This returns a `Iterator[Record]` which is an
365        iterator.
366
367        Streaming is one of the two ways to consume events in Fluvio. It is a
368        continuous request for new records arriving in a partition, beginning
369        at a particular offset. You specify the starting point of the stream
370        using an Offset and periodically receive events, either individually or
371        in batches.
372        """
373        return self._generator(self._inner.stream(offset))
374
375    async def async_stream(self, offset: Offset) -> typing.AsyncIterator[Record]:
376        """
377        Continuously streams events from a particular offset in the consumer’s
378        partition. This returns a `AsyncIterator[Record]` which is an
379        iterator.
380

382        continuous request for new records arriving in a partition, beginning
383        at a particular offset. You specify the starting point of the stream
384        using an Offset and periodically receive events, either individually or
385        in batches.
386        """
387        return self._async_generator(await self._inner.async_stream(offset))

453    def _generator(self, stream: _PartitionConsumerStream) -> typing.Iterator[Record]:
454        item = stream.next()
455        while item is not None:
456            yield Record(item)
457            item = stream.next()
458
459    async def _async_generator(
460        self, astream: _AsyncPartitionConsumerStream
461    ) -> typing.AsyncIterator[Record]:
462        item = await astream.async_next()
463        while item is not None:
464            yield Record(item)
465            item = await astream.async_next()
