187        Args:

211            raise Exception("Only specify one of path or name not both.")
212
213        params = {} if params is None else params
214        param_keys = [x for x in params.keys()]
215        param_vals = [x for x in params.values()]
216
217        self._inner.smartmodule(
218            name,
219            path,
220            kind,
221            param_keys,
222            param_vals,
223            aggregate,

227            None,

233class PartitionConsumer:
234    """
235    An interface for consuming events from a particular partition
236
237    There are two ways to consume events: by "fetching" events and by
238    "streaming" events. Fetching involves specifying a range of events that you
239    want to consume via their Offset. A fetch is a sort of one-time batch
240    operation: youll receive all of the events in your range all at once. When
241    you consume events via Streaming, you specify a starting Offset and receive
242    an object that will continuously yield new events as they arrive.
243    """
244

246

250    def stream(self, offset: Offset) -> typing.Iterator[Record]:
251        """
252        Continuously streams events from a particular offset in the consumers
253        partition. This returns a `Iterator[Record]` which is an
254        iterator.
255
256        Streaming is one of the two ways to consume events in Fluvio. It is a
257        continuous request for new records arriving in a partition, beginning
258        at a particular offset. You specify the starting point of the stream
259        using an Offset and periodically receive events, either individually or
260        in batches.
261        """
262        return self._generator(self._inner.stream(offset._inner))

493class TopicProducer:
494    """An interface for producing events to a particular topic.
495
496    A `TopicProducer` allows you to send events to the specific topic it was
497    initialized for. Once you have a `TopicProducer`, you can send events to
498    the topic, choosing which partition each event should be delivered to.
499    """
500
501    _inner: _TopicProducer
502
503    def __init__(self, inner: _TopicProducer):
504        self._inner = inner
505
506    def send_string(self, buf: str) -> ProduceOutput:
507        """Sends a string to this producers topic"""
508        return self.send([], buf.encode("utf-8"))
509
510    async def async_send_string(self, buf: str) -> ProduceOutput:
511        """Sends a string to this producers topic"""
512        return await self.async_send([], buf.encode("utf-8"))
513
514    def send(self, key: typing.List[int], value: typing.List[int]) -> ProduceOutput:
515        """
516        Sends a key/value record to this producer's Topic.
517
518        The partition that the record will be sent to is derived from the Key.
519        """
520        return ProduceOutput(self._inner.send(key, value))
521
522    async def async_send(
523        self, key: typing.List[int], value: typing.List[int]
524    ) -> ProduceOutput:
525        """
526        Async sends a key/value record to this producer's Topic.
527
528        The partition that the record will be sent to is derived from the Key.
529        """
530        produce_output = await self._inner.async_send(key, value)
531        return ProduceOutput(produce_output)
532
533    def flush(self) -> None:
534        """
535        Send all the queued records in the producer batches.
536        """
537        return self._inner.flush()
538
539    async def async_flush(self) -> None:
540        """
541        Async send all the queued records in the producer batches.
542        """
543        await self._inner.async_flush()
544
545    def send_all(
546        self, records: typing.List[typing.Tuple[bytes, bytes]]
547    ) -> typing.List[ProduceOutput]:
548        """
549        Sends a list of key/value records as a batch to this producer's Topic.
550        :param records: The list of records to send
551        """
552        records_inner = [_ProducerBatchRecord(x, y) for (x, y) in records]
553        return [
554            ProduceOutput(output_inner)
555            for output_inner in self._inner.send_all(records_inner)
